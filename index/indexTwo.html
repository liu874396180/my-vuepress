# 用户信息 userInfo API

	//1. 获取用户ID:
	this.H5Plus.userInfo.get("ID");
	//2. 获取用户TOKEN:
	this.H5Plus.userInfo.get("TOKEN");
	//3. 获取用户手机号:
	this.H5Plus.userInfo.get("PHONE_NUM");
	//4. 获取用户渠道:
	this.H5Plus.userInfo.get("CHANNEL");
	//5. 获取用户渠道ID:
	this.H5Plus.userInfo.get("CHANNEL_ID");
	//6. 获取用户OPEN_API_CHANNEL_ID:
	this.H5Plus.userInfo.get("OPEN_API_CHANNEL_ID");
	//7. 获取会话ID:
	this.H5Plus.userInfo.get("SESSION_ID");
	//8. 获取APP_FLAG:
	this.H5Plus.userInfo.get("APP_FLAG");
	//9. 获取VERSION:
	this.H5Plus.userInfo.get("VERSION");
	//10. 获取SYSTEM_TYPE:
	this.H5Plus.userInfo.get("SYSTEM_TYPE");
	//11. 获取NETWORK_TYPE:
	this.H5Plus.userInfo.get("NETWORK_TYPE");
	//12. 获取设备ID:
	this.H5Plus.userInfo.get("DEVICE_ID");
	//13. 获取设备名称:
	this.H5Plus.userInfo.get("DEVICE_NAME");

# 浏览器信息 browserInfo API


	//1. 获取浏览器UA:
	this.H5Plus.browserInfo.get("UA")

	//2. 获取浏览器类型:
	this.H5Plus.browserInfo.get("BROWSER")

	//3. 获取浏览器版本:
	this.H5Plus.browserInfo.get("BROWSER_VERSION")

	//4. 获取浏览器名称:
	this.H5Plus.browserInfo.get("BROWSER_NAME")

	//5. 获取系统类型:
	this.H5Plus.browserInfo.get("OS")

	//6. 获取系统版本:
	this.H5Plus.browserInfo.get("OS_VERSION")

	//7. 获取网络类型:
	this.H5Plus.browserInfo.get("NETWORK_TYPE")

# 请求 httpPost API

	// 已封装比财api请求体 已处理响应拦截器
	this.H5Plus.httpPost(
		"https://finsuit.bicai365.com",
		"/finsuit/finsuitPhone/deal",
		"GET_WEIXIN_SIGN",
		{},
		{},
		false,
		false);

# 请求 http API

	// 未封装比财api请求体 已处理响应拦截器
	this.H5Plus.http({
		method: "post",
		url: "/finsuit/finsuitPhone/deal",
		baseURL: "https://finsuit.bicai365.com",
		data: {},
		timeout: 20000,
		isLoading: false,
		isGetCode: false,
		headers: {
			"Content-Type": "application/x-www-form-urlencoded;charset=UTF-8"
		}
	});
